  <script>

    // Smooth anchor scroll

    document.querySelectorAll('a[href^="#"]').forEach(a => {

      a.addEventListener('click', e => {

        const id = a.getAttribute('href');

        if (!id || id === '#') return;

        const el = document.querySelector(id);

        if (!el) return;

        e.preventDefault();

        el.scrollIntoView({ behavior: 'smooth', block: 'start' });

        document.getElementById('mobile-menu').classList.remove('mobile--open');

      });

    });


    // Mobile menu

    document.getElementById('mobile-menu-btn').addEventListener('click', () => {

      document.getElementById('mobile-menu').classList.toggle('mobile--open');

    });


    // CTA buttons -> scroll to contact

    function scrollToContact() {

      document.getElementById('contact').scrollIntoView({ behavior: 'smooth', block: 'start' });

    }

    document.getElementById('cta-top').addEventListener('click', scrollToContact);

    document.getElementById('cta-bottom').addEventListener('click', scrollToContact);


    // Navbar shadow on scroll

    const nav = document.querySelector('.nav');

    window.addEventListener('scroll', () => {

      if (window.scrollY > 50) nav.classList.add('nav--scrolled');

      else nav.classList.remove('nav--scrolled');

    });


    // Contact form -> mailto

    document.getElementById('contact-form').addEventListener('submit', function(e){

      e.preventDefault();

      const data = new FormData(this);

      const name = data.get('name');

      const email = data.get('email');

      const company = data.get('company') || 'Not provided';

      const message = data.get('message');


      const subject = encodeURIComponent(`New inquiry from ${name}${company && company !== 'Not provided' ? ' at ' + company : ''}`);

      const body = encodeURIComponent(

        `Name: ${name}\nEmail: ${email}\nCompany: ${company}\n\nMessage:\n${message}`

      );


      window.location.href = `mailto:yamenalshdifat@gmail.com?subject=${subject}&body=${body}`;

      alert("Thank you for your message! Your email client should open with a pre-filled message. If not, please email us directly at yamenalshdifat@gmail.com");

      this.reset();

    });

  </script>
